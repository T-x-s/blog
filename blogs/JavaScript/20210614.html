    这是自己总结的 JavaScript 的一些常用方法

    # 1、字符串对象的方法

    s (1)length获取字符串的长度

    let str = "Hello world!"
let str1 = "wo";
console.log(str.length);  // 12
    1
    2
    3

    (2)charAt(下标)获取对应位置的字符

    console.log(str.charAt(0));  // H
    1

    (3)indexOf(str[,index])返回某个指定的字符串值在字符串中首次出现的位置

    console.log(str.indexOf(str1));  // 6 万一查找到某个字符在字符串中不存在，返回-1
    1

    (4)lastIndexOf(str[,index])返回某个指定的字符串值在字符串中最后一次出现的位置

    console.log(str.lastIndexOf(str1));  // 6
    1

    (5)substring([start,[stop]])截取字符串,从 start 到 stop,包括起点不包括终点

    console.log(str.substring(0,2));  // He
console.log(str.substring(-5,1));  // H

    2

    (6)slice([start,[stop]])截取字符串

    console.log(str.slice(2,5))  // llo 返回索引2-5但不包含5之间的字符
console.log(str.slice(2,-5)) // llo w 返回从索引2开始至倒数第五个位置但不包含其本身之间的字符。
console.log(str.slice(-5,-1)) // orld 返回倒数第五个位置至倒数第一个位置(不包含倒数第一个位置)之间的字符，且两个参数均为负数时，第二个参数必须大于第一个参数

    ①slice和substring的区别
    slice与substring的区别在于当接收的参数是负数时，slice会将它字符串的长度与对应的负数相加，结果作为参数。substring则干脆将负参数都直接转换为 0
    当slice和substring的参数传入顺序颠倒，substring可以将错误自行纠正，而slice不行

    (7)toUpperCase转换为大写 字符串.toUpperCase()

    console.log(str.toUpperCase());  // HELLO WORLD!
    1

    (8)toLowerCase转换为小写 字符串.toLowerCase()

    console.log(str.toLowerCase());  // hello world!
    1

    (9)split(分割标准，[数组的长度]) ： 将字符串按特定标志分割成数组，且数组长度为参数规定的长度

    console.log(str.split(""));  // ["H", "e", "l", "l", "o", " ", "w", "o", "r", "l", "d", "!"]
    1

    (10)replace（被替换的字符串，替换的字符）: 替换特定的字符

     console.log(str.replace('H','h')); // hello world!
    1

    (11)search()用于检索字符串中指定的子字符串

    console.log(str.search('He'));// 0
    1

    (12)match()可在字符串内检索指定的值，或找到一个或多个正则表达式的匹配

    console.log(str2.match('He')); // ["He", index: 0, input: "Hello world!", groups: undefined]  返回值是一个数组，数组中会有要查找的值对应的相关信息   注意：只获取第一个匹配项
console.log(str2.match('llo')); // ["llo", index: 2, input: "Hello world!", groups: undefined]
    1
    2

    (13)substr()可在字符串中抽取从 start 下标开始的指定数目的字符。

    console.log(str.substr(2,5))  // llo w 从索引2开始，返回5个字符
console.log(str.substr(-2,5))  // d! 即从倒数第二个字符开始返回5个字符，第一个参数可为负值，但第二个参数不可为负值

    2

    注：字符串对象的所有方法都不会改变原字符串，而是生成新的字符串

    14)trim()可以去除字符串两边的空格，常用于提交表单。中间空格会被认为是有价值的，不会被删除。

    console.log(xxx.trim());

console.log(arr.splice(0,1)); // ["abc"]
console.log(arr); // []

// 替换用法：a)数组 . splice（开始删除的索引，要删除的项数，要添加的数组项1,要添加的数组项2...）
// 原理 1、删除要被替换的数组项 2、ii.将要替换的数组项添加到原来被替换的数组项的位置

console.log(arr); // ["abc"]

// 添加用法：a)数组.splice（开始删除的索引，0，要添加的数组项1，要添加的数组项2....）

// 注意：如果向在第n项后面添加数组项，一定要将开始删除的索引设置为n+1

console.log(arr); // ["abc", "abc"]

    # 5、Object 方法

    (1)Object.keys()方法会返回一个由一个给定对象的自身可枚举属性组成的数组，数组中属性名的排列顺序和正常循环遍历该对象时返回的顺序一致 。

    var arr = ['a', 'b', 'c'];
console.log(Object.keys(arr)); // console: ['0', '1', '2']
    1
    2

    (2)Object.assign()(ES6 新增浅拷贝) 方法用于将所有可枚举属性的值从一个或多个源对象分配到目标对象。它将返回目标对象。

    const target = { a: 1, b: 2 };
const source = { b: 4, c: 5 };
const returnedTarget = Object.assign(target, source);
console.log(target);
// expected output: Object { a: 1, b: 4, c: 5 }
console.log(returnedTarget);
// expected output: Object { a: 1, b: 4, c: 5 }
